### 步骤 2：重写 onLoad 方法

- [ ] **更新事件绑定**

将原来的按钮事件绑定替换为滑条事件绑定：

```typescript
onLoad() {
    // 设置滑条事件
    if (this.gridSizeSlider) {
        this.gridSizeSlider.node.on(Slider.EventType.SLIDE, this.onSliderChanged, this);
    }
    
    // 设置开始按钮事件
    if (this.startButton) {
        this.startButton.node.on(Node.EventType.TOUCH_END, this.onStart, this);
    }
    // 设置取消按钮事件
    if (this.cancelButton) {
        this.cancelButton.node.on(Node.EventType.TOUCH_END, this.onCancel, this);
    }
}
```

### 步骤 3：重写 show 方法

- [ ] **更新 show 方法实现**

```typescript
show(currentSize: GridSize, onConfirm: (size: GridSize) => void): void {
    this._selectedSize = currentSize;
    this._onConfirm = onConfirm;
    
    // 设置滑条初始值 (4-8 映射到 0-4)
    if (this.gridSizeSlider) {
        this.gridSizeSlider.progress = (currentSize - 4) / 4;
    }
    
    // 更新数值显示
    this.updateValueLabel();
    
    this.node.active = true;
}
```

### 步骤 4：添加滑条事件处理方法

- [ ] **添加 onSliderChanged 方法**

```typescript
private onSliderChanged(): void {
    if (!this.gridSizeSlider) return;
    
    // 将滑条进度 (0-1) 映射到网格大小 (4-8)
    const rawValue = this.gridSizeSlider.progress * 4; // 0-4
    const index = Math.round(rawValue);
    this._selectedSize = GRID_SIZES[index];
    
    // 吸附到刻度
    this.gridSizeSlider.progress = index / 4;
    
    // 更新显示
    this.updateValueLabel();
}
```

- [ ] **添加 updateValueLabel 方法**

```typescript
private updateValueLabel(): void {
    if (this.gridSizeValueLabel) {
        this.gridSizeValueLabel.string = this._selectedSize.toString();
    }
}
```

### 步骤 5：移除旧方法

- [ ] **删除以下不再需要的方法**

- `updateButtonVisuals()`

## 任务 2：创建 GameOverPanel.ts

**文件：**
- 创建：`assets/games/game_2048/scripts/GameOverPanel.ts`

### 步骤 1：创建文件并添加导入

- [ ] **创建文件并写入基础结构**

```typescript
import { _decorator, Component, Node, Button, Label } from 'cc';

const { ccclass, property } = _decorator;

@ccclass('GameOverPanel')
export class GameOverPanel extends Component {
    @property(Label)
    scoreValueLabel: Label | null = null;
    
    @property(Button)
    restartButton: Button | null = null;
    
    @property(Button)
    backButton: Button | null = null;
    
    private _onRestart: (() => void) | null = null;
    private _onBack: (() => void) | null = null;
}
```

### 步骤 2：实现 onLoad 方法

- [ ] **添加事件绑定**

```typescript
onLoad() {
    // 绑定重新开始按钮
    if (this.restartButton) {
        this.restartButton.node.on(Node.EventType.TOUCH_END, this.onRestartClick, this);
    }
    
    // 绑定返回大厅按钮
    if (this.backButton) {
        this.backButton.node.on(Node.EventType.TOUCH_END, this.onBackClick, this);
    }
    
    // 默认隐藏
    this.node.active = false;
}
```

### 步骤 3：实现 show 方法

- [ ] **添加 show 方法**

```typescript
show(score: number, onRestart: () => void, onBack: () => void): void {
    this._onRestart = onRestart;
    this._onBack = onBack;
    
    // 更新分数显示
    if (this.scoreValueLabel) {
        this.scoreValueLabel.string = score.toString();
    }
    
    // 显示面板
    this.node.active = true;
}
```

### 步骤 4：实现 hide 方法

- [ ] **添加 hide 方法**

```typescript
hide(): void {
    this.node.active = false;
    this._onRestart = null;
    this._onBack = null;
}
```

### 步骤 5：实现按钮回调

- [ ] **添加按钮回调方法**

```typescript
private onRestartClick(): void {
    this.hide();
    this._onRestart?.();
}

private onBackClick(): void {
    this.hide();
    this._onBack?.();
}
```
